# here we get rid of that added dimension and plot the image
def visualize_img(model, img):
    # Keras expects batches of images, so we have to add a dimension to trick it into being nice
    img_batch = np.expand_dims(img,axis=0)
    conv_img = model.predict(img_batch)
    conv_img = np.squeeze(conv_img, axis=0)
    print (conv_img.shape)
    plt.imshow(conv_img)

def nice_img_printer(model, img):
    '''prints the img as a 2d array'''
    img_batch = np.expand_dims(img,axis=0)

    conv_img2 = np.squeeze(conv_img2, axis=0)

    conv_img2 = conv_img2[:,:,0]

def preprocess_image(x,v2=True):
    """
    scale the image to [0,1] for CNN
    (v2 scale the image to [-1,1] better range for CNN)
    """
    x = x.astype('float32')
    x = x/255
    if v2:
        x -= 0.5
        x *= 2.0
    return x
